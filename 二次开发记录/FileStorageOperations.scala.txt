  /**
   * 得到File by Lining 2016-6-28
   * @param fileId
   * @param fileName
   * @return
   */
  protected def getFile(fileId: Long, fileName: String): Future[Option[File]] =
    db.run(FileRepo.find(fileId)) map {
      case Some(model) ⇒
        val name = if (fileName.trim.isEmpty) model.name else fileName
        Some(fileDirectory(fileId) / name)
      case None ⇒ None
    }
